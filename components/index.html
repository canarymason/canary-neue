Accessibility

Sass Mixin Source 
//////////////////////////////
// Accessibility Component Mixins
//////////////////////////////

//////////////////////////////
// A11y Default Variables
$a11y-hidden-base-extend: true !default;
$a11y-focusable-default: false !default;
//////////////////////////////

@mixin a11y-hidden($a11y-focusable: $a11y-focusable-default, $extend: $a11y-hidden-base-extend) {
  @if $extend {
    @if $a11y-focusable {
      @extend %a11y-hidden-focusable-extend;
    }
    @else {
      @extend %a11y-hidden-base-extend;
    }
  }
  @else {
    border: 0;
    clip: rect(0 0 0 0);
    height: 1px;
    margin: -1px;
    overflow: hidden;
    padding: 0;
    position: absolute;
    width: 1px;
    @if $a11y-focusable {
      &:active, &:focus {
        clip: auto;
        height: auto;
        margin: 0;
        overflow: visible;
        position: static;
        width: auto;
      }
    }
  }
}
Sass Extend Source 
//////////////////////////////
// Accessibility Component Extendable Classes
//////////////////////////////

$a11y-extendables-extended: false !default;

@if not ($a11y-extendables-extended) {
  %hidden {
    display: none;
  }

  %a11y-hidden-base-extend {
    @include a11y-hidden($extend: false);
  }

  %a11y-hidden-focusable-extend {
    @include a11y-hidden($a11y-focusable: true, $extend: false);
  }
}

Sass Mixin Source 
//////////////////////////////
// Message Component Mixins
//////////////////////////////

//////////////////////////////
// Messages Component Default Variables
$message-base-extend: true !default;
$message-colors-extend: true !default;
$message-color-adjust: 10% !default;
$message-text-color-default: $cs-message-text-color-default !default;
$message-bkg-color-default: $cs-message-bkg-color-default !default; // false
//////////////////////////////

@mixin message-base($extend: $message-base-extend) {
  @if $extend {
    @extend %message-base-extend;
  }
  @else {
    padding: 0.2em 0.5em;
    border: {
      style: solid;
      width: .15em;
      radius: .25em;
    }
  }
}

@mixin message-colors($message-color: $message-bkg-color-default, $message-adjust: $message-color-adjust,  $extend: $message-colors-extend) {
  @if ($message-color != $message-bkg-color-default) {
    $extend: false;
  }
  @if $extend {
    @extend %message-colors-extend;
  }
  @else {
    @include readable-text($message-color);

    @if cs-is-dark($message-color) {
      background-color: blend-screen($message-color, adjust-color($message-color, $lightness: $message-adjust * -1));

      border-color: blend-screen($message-color, adjust-color($message-color, $lightness: $message-adjust * -2.5));
    }
    @else {
      background-color: blend-multiply($message-color, adjust-color($message-color, $lightness: $message-adjust * -1));

      border-color: blend-multiply($message-color, adjust-color($message-color, $lightness: $message-adjust * -2.5));
    }
  }
}
Sass Extend Source 
//////////////////////////////
// Message Component Extendable Classes
//////////////////////////////

$message-extendables-extended: false !default;

@if not ($message-extendables-extended) {
  %message-base-extend {
    @include message-base(false);
  }
  %message-colors-extend {
    @include message-colors($cs-status-status, $extend: false);
  }

  %message {
    @extend %message-base-extend;
    @extend %message-colors-extend;
  }
  %status {
    @include message-colors($cs-status-status);
  }
  %warning {
    @include message-colors($cs-status-warning);
  }
  %error {
    @include message-colors($cs-status-error);
  }
}

@mixin pager-container($extend: $pager-container-extend) {
  @if $extend {
    @extend %pager-container-extend;
  }
  @else {
    display: inline;
    margin: 0;
    padding: 0;
    list-style-type: none;
  }
}

@mixin pager-item($extend: $pager-item-extend) {
  @if $extend {
    @extend %pager-item-extend;
  }
  @else {
    display: inline;
    margin: 0;
    padding: 0.2em 0.5em;
    text-transform: uppercase;
  }
}
